PDFGEN_SCRIPT = path.join(__dirname, '..', 'generate-pdf.sh')

childLogger = (promise) -> (data) ->
  lines = data.toString().split('\n')
  for line in lines
    if line.length > 1
      percent = /^STATUS: (\d+)%/.exec(line)
      if percent
        promise.progress.finished = parseInt(percent[1])
      else
        promise.update line
